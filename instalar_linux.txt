#!/bin/bash

# Script de instalación web para Calculadora Profesional
# INSTRUCCIONES: Renombra este archivo a "instalar_linux.sh" y ejecuta:
# chmod +x instalar_linux.sh && ./instalar_linux.sh

GREEN='\033[0;32m'
BLUE='\033[0;34m'
RED='\033[0;31m'
YELLOW='\033[1;33m'
NC='\033[0m'

clear
echo -e "${BLUE}=========================================="
echo -e "  INSTALADOR WEB AUTOMÁTICO"
echo -e "  CALCULADORA PROFESIONAL v2.0"
echo -e "==========================================${NC}"
echo ""
echo -e "${YELLOW}Este script instalará automáticamente la calculadora${NC}"
echo ""

# Verificar permisos
if [ "$EUID" -eq 0 ]; then 
    echo -e "${RED}No ejecutes este script como root/sudo${NC}"
    echo "Se instalará en tu directorio personal"
    exit 1
fi

# Verificar Python3
echo -e "${BLUE}[1/6] Verificando Python3...${NC}"
if ! command -v python3 &> /dev/null; then
    echo -e "${RED}✗ Python3 no está instalado${NC}"
    echo ""
    echo "Por favor instala Python3 primero:"
    echo "  Ubuntu/Debian: sudo apt install python3 python3-tk"
    echo "  Fedora: sudo dnf install python3 python3-tkinter"
    echo "  Arch: sudo pacman -S python tk"
    exit 1
fi
echo -e "${GREEN}✓ Python3 encontrado${NC}"

# Verificar tkinter
echo -e "${BLUE}[2/6] Verificando tkinter...${NC}"
if ! python3 -c "import tkinter" 2>/dev/null; then
    echo -e "${YELLOW}⚠ tkinter no está instalado${NC}"
    echo -e "${BLUE}Intentando instalar tkinter...${NC}"
    
    # Detectar distribución
    if [ -f /etc/os-release ]; then
        . /etc/os-release
        case $ID in
            ubuntu|debian|linuxmint)
                sudo apt update && sudo apt install python3-tk -y
                ;;
            fedora|rhel|centos)
                sudo dnf install python3-tkinter -y
                ;;
            arch|manjaro)
                sudo pacman -S tk --noconfirm
                ;;
            *)
                echo -e "${RED}Distribución no reconocida. Instala tkinter manualmente.${NC}"
                exit 1
                ;;
        esac
    fi
    
    # Verificar de nuevo
    if ! python3 -c "import tkinter" 2>/dev/null; then
        echo -e "${RED}✗ No se pudo instalar tkinter${NC}"
        exit 1
    fi
fi
echo -e "${GREEN}✓ tkinter disponible${NC}"

# Crear directorios
echo -e "${BLUE}[3/6] Creando directorios...${NC}"
INSTALL_DIR="$HOME/.local/share/CalculadoraProfesional"
DESKTOP_DIR="$HOME/.local/share/applications"
mkdir -p "$INSTALL_DIR"
mkdir -p "$DESKTOP_DIR"
echo -e "${GREEN}✓ Directorios creados${NC}"

# Descargar/Crear archivo Python
echo -e "${BLUE}[4/6] Instalando aplicación...${NC}"

# Aquí puedes poner la URL de descarga o embeber el código
# Por ahora, verificamos si existe localmente
if [ -f "calculadora_completa.py" ]; then
    cp calculadora_completa.py "$INSTALL_DIR/"
    echo -e "${GREEN}✓ Archivo copiado desde directorio local${NC}"
else
    # Si no existe, mostrar mensaje
    echo -e "${YELLOW}⚠ Archivo calculadora_completa.py no encontrado${NC}"
    echo -e "${BLUE}Descarga el paquete completo desde la web${NC}"
    exit 1
fi

# Copiar icono si existe
if [ -f "icono.ico" ]; then
    cp icono.ico "$INSTALL_DIR/"
fi

# Crear script ejecutable
echo -e "${BLUE}[5/6] Configurando ejecutable...${NC}"
cat > "$INSTALL_DIR/calculadora.sh" << 'EOF'
#!/bin/bash
cd "$(dirname "$0")"
python3 calculadora_completa.py
EOF

chmod +x "$INSTALL_DIR/calculadora.sh"
echo -e "${GREEN}✓ Ejecutable configurado${NC}"

# Crear entrada en el menú
echo -e "${BLUE}[6/6] Creando accesos directos...${NC}"
cat > "$DESKTOP_DIR/calculadora-profesional.desktop" << EOF
[Desktop Entry]
Version=2.0
Type=Application
Name=Calculadora Profesional
Comment=Calculadora científica con soporte para física
Exec=$INSTALL_DIR/calculadora.sh
Icon=$INSTALL_DIR/icono.ico
Terminal=false
Categories=Utility;Calculator;Science;Education;
Keywords=calculator;math;science;physics;
StartupNotify=true
EOF

chmod +x "$DESKTOP_DIR/calculadora-profesional.desktop"

# Actualizar base de datos de aplicaciones
update-desktop-database "$DESKTOP_DIR" 2>/dev/null

# Crear desinstalador
cat > "$INSTALL_DIR/desinstalar.sh" << 'UNINSTALL_EOF'
#!/bin/bash
rm -rf "$HOME/.local/share/CalculadoraProfesional"
rm -f "$HOME/.local/share/applications/calculadora-profesional.desktop"
rm -f "$HOME/Desktop/calculadora-profesional.desktop"
rm -f "$HOME/Escritorio/calculadora-profesional.desktop"
update-desktop-database "$HOME/.local/share/applications" 2>/dev/null
echo "Calculadora Profesional desinstalada"
UNINSTALL_EOF

chmod +x "$INSTALL_DIR/desinstalar.sh"

echo -e "${GREEN}✓ Accesos directos creados${NC}"

# Mensaje final
echo ""
echo -e "${GREEN}=========================================="
echo -e "  ✓ INSTALACIÓN COMPLETADA"
echo -e "==========================================${NC}"
echo ""
echo -e "${YELLOW}La calculadora ha sido instalada en:${NC}"
echo "  $INSTALL_DIR"
echo ""
echo -e "${YELLOW}Para ejecutar:${NC}"
echo "  • Busca 'Calculadora Profesional' en tu menú de aplicaciones"
echo "  • O ejecuta: $INSTALL_DIR/calculadora.sh"
echo ""
echo -e "${YELLOW}Para desinstalar:${NC}"
echo "  $INSTALL_DIR/desinstalar.sh"
echo ""
echo -e "${GREEN}¡Disfruta tu calculadora científica!${NC} 🧮"
echo ""
